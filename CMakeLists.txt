cmake_minimum_required(VERSION 2.6)

PROJECT(teng)

FIND_PACKAGE(OpenThreads)
FIND_PACKAGE(osg)
FIND_PACKAGE(osgDB)
FIND_PACKAGE(osgViewer)
FIND_PACKAGE(osgShadow)
FIND_PACKAGE(osgSim)
FIND_PACKAGE(osgParticle)
FIND_PACKAGE(osgTerrain)
FIND_PACKAGE(osgFX)

SET(SOURCES
    qsc.cpp kml.cpp Patch.cpp PatchGroup.cpp PatchSet.cpp patchTest.cpp
)

INCLUDE_DIRECTORIES(${OPENTHREADS_INCLUDE_DIR} ${OSG_INCLUDE_DIR})

LINK_DIRECTORIES(${OSG_LIB_DIR})

ADD_EXECUTABLE(kmld kml.cpp qsc.cpp)

ADD_EXECUTABLE(patchTest patchTest.cpp Patch.cpp PatchGroup.cpp PatchSet.cpp)

TARGET_LINK_LIBRARIES(kmld ${OSG_LIBRARY_DEBUG})

TARGET_LINK_LIBRARIES(patchTest ${OSG_LIBRARY_DEBUG} ${OSGVIEWER_LIBRARY_DEBUG} ${OSGSIM_LIBRARY_DEBUG} ${OSGTERRAIN_LIBRARY_DEBUG} ${OSGFX_LIBRARY_DEBUG} ${OSGPARTICLE_LIBRARY_DEBUG} ${OSGSHADOW_LIBRARY_DEBUG} ${OPENTHREADS_LIBRARY_DEBUG})
